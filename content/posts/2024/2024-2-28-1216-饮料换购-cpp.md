# 2024-2-28-1216- 饮料换购.cpp
```cpp
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
int res;
int main()
{
    int n;
    cin>>n;
    int res=n;
    while(n>=3)
    {
      res+=n/3;  
    n=n/3+n%3;
    }
    cout<<res;
    return 0;
    
}
```
