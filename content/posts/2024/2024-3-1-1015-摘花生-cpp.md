# 2024-3-1-1015-摘花生.cpp

```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N =105;
int res;
int T,R,C,M[N][N],F[N][N];
int main()
{
cin>>T;
while(T--)
{
scanf("%d%d",&R,&C);
for(int i=1;i<=R;i++)
    for(int j=1;j<=C;j++)
        scanf("%d",&M[i][j]);
for(int i=1;i<=R;i++)
    for(int j=1;j<=C;j++)
       {F[i][j]=M[i][j];
        F[i][j]+=max(F[i-1][j],F[i][j-1]);
}cout<<F[R][C]<<endl;

}
    return 0;
}
```
