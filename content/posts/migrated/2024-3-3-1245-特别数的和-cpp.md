# 2024-3-3-1245- 特别数的和.cpp

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>
#include<cstdio>
int n,res;
using namespace std;
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    {int j=i;
        while(j)
        {
            int cnt=j%10;
            j=j/10;
            if(cnt==2||cnt==0||cnt==1||cnt==9)
            {res+=i;break;}
        }
    }
    cout<<res;
    return 0;
}
```
